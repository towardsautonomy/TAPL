24 std::vector<Eigen::MatrixXd> computeInitialRTfromE(const Eigen::MatrixXd &E) {
25 
26  // SVD decomposition
27  Eigen::JacobiSVD<Eigen::MatrixXd> svd(E, Eigen::ComputeThinU | Eigen::ComputeThinV);
28 
29  // build 'Z' and 'W'
30  Eigen::MatrixXd Z(3,3);
31  Z << 0, 1, 0,
32  -1, 0, 0,
33  0, 0, 0;
34  Eigen::MatrixXd W(3,3);
35  W << 0, -1, 0,
36  1, 0, 0,
37  0, 0, 1;
38 
39  // E can be re-written as E=MQ; where M=U.Z.U^T and Q=U.W.V^T or Q=U.W^T.V^T
40  Eigen::MatrixXd M = svd.matrixU() * Z * svd.matrixU().transpose();
41  Eigen::MatrixXd Q1 = svd.matrixU() * W * svd.matrixV().transpose();
42  Eigen::MatrixXd Q2 = svd.matrixU() * W.transpose() * svd.matrixV().transpose();
43 
44  // R can be computed as R=(det Q)Q
45  Eigen::MatrixXd R1 = Q1.determinant() * Q1;
46  Eigen::MatrixXd R2 = Q2.determinant() * Q2;
47 
48  // E = UΣV^T, T is simply either u3 or −u3, where u3 is the third column vector of U
49  Eigen::MatrixXd T1 = svd.matrixU().col(2);
50  Eigen::MatrixXd T2 = -svd.matrixU().col(2);
51 
52  // compose four possible RT
53  std::vector<Eigen::MatrixXd> RT;
54  // R1 and T1
55  Eigen::MatrixXd R1T1(3,4);
56  R1T1.block<3,3>(0,0) = R1;
57  R1T1.block<3,1>(0,3) = T1;
58  RT.push_back(R1T1);
59  // R1 and T2
60  Eigen::MatrixXd R1T2(3,4);
61  R1T2.block<3,3>(0,0) = R1;
62  R1T2.block<3,1>(0,3) = T2;
63  RT.push_back(R1T2);
64  // R2 and T1
65  Eigen::MatrixXd R2T1(3,4);
66  R2T1.block<3,3>(0,0) = R2;
67  R2T1.block<3,1>(0,3) = T1;
68  RT.push_back(R2T1);
69  // R2 and T2
70  Eigen::MatrixXd R2T2(3,4);
71  R2T2.block<3,3>(0,0) = R2;
72  R2T2.block<3,1>(0,3) = T2;
73  RT.push_back(R2T2);
74 
75  // return
76  return RT;
77 }

86  // by decomposing A using SVD
87  // let's formulate the A matrix
88  Eigen::MatrixXd A(point2d.size()*2, 4);
89  for (auto i=0; i<point2d.size(); ++i) {
90  A.row(i*2) << (point2d.at(i).x * projectionMatrices.at(i).row(2)) -
91  projectionMatrices.at(i).row(0);
92  A.row((i*2)+1) << (point2d.at(i).y * projectionMatrices.at(i).row(2)) -
93  projectionMatrices.at(i).row(1);
94  }
95  // SVD decomposition
96  Eigen::JacobiSVD<Eigen::MatrixXd> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);

98  Eigen::MatrixXd pt3d_homogeneous = svd.matrixV().col(3);
99  // homogeneous to cartesian coordinate conversion
100  tapl::Point3d pt3d( pt3d_homogeneous(0,0) / pt3d_homogeneous(3,0),
101  pt3d_homogeneous(1,0) / pt3d_homogeneous(3,0),
