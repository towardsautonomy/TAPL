108  {
109

114  }

119  int apertureSize = 3; // aperture parameter for Sobel operator (must be odd)

121  double k = 0.04; // Harris parameter (see equation for details)

131  for (size_t j = 0; j < dst_norm.rows; j++) {
132  for (size_t i = 0; i < dst_norm.cols; i++) {
133  int response = (int)dst_norm.at<float>(j, i);
134  if (response > minResponse) { // only store points above a threshold
135

137  newKeyPoint.pt = cv::Point2f(i, j);
138  newKeyPoint.size = 2 * apertureSize;

274 
275  // create a copy of the descriptors so that any modification does not effect rest of the pipeline
276  cv::Mat descQryCopy, descTrnCopy;
277  descQuery.copyTo(descQryCopy);
278  descRef.copyTo(descTrnCopy);
279

283  }

285  if (descQryCopy.type() != CV_32F) { // OpenCV bug workaround : convert binary descriptors to floating point due to a bug in current OpenCV implementation
286  descQryCopy.convertTo(descQryCopy, CV_32F);
287  descTrnCopy.convertTo(descTrnCopy, CV_32F);
288  }
289 
290  matcher = cv::DescriptorMatcher::create(cv::DescriptorMatcher::FLANNBASED);
291  }
292 
293  // perform matching task
294  if (selectorType.compare("SEL_NN") == 0) { // nearest neighbor (best match)
295 
296  matcher->match(descQryCopy, descTrnCopy, matches); // Finds the best match for each descriptor in desc1
297  }
298  else if (selectorType.compare("SEL_KNN") == 0) {
299  std::vector<std::vector<cv::DMatch>> knn_matches;
300  matcher->knnMatch(descQryCopy, descTrnCopy, knn_matches, 2); // finds the 2 best matches
301

303  double minDescDistRatio = 0.8;
304  for (auto it = knn_matches.begin(); it != knn_matches.end(); ++it) {
305 
306  if ((*it)[0].distance < minDescDistRatio * (*it)[1].distance) {
307  matches.push_back((*it)[0]);
308  }
309  }
310 
311  }
312  else {
313  // use nearest-neighbor method as default
314  matcher->match(descQryCopy, descTrnCopy, matches); // Finds the best match for each descriptor in desc1
315  }
316
