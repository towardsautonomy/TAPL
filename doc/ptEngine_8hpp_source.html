150  cv::Mat world2CamRotation()
151  {
152  // camera coordinate to world coordinate rotation matrix
153  cv::Mat R = cv::Mat::zeros(3, 3, CV_32F);
154  // Camera rotation
155  float Rx = degreesToRadians(-90);
156  float Ry = degreesToRadians(0);
157  float Rz = degreesToRadians(-90);
158 
159  // Rz
160  cv::Mat R_z = cv::Mat::eye(3, 3, CV_32F);
161  R_z.at<float>(0, 0) = cos(Rz);
162  R_z.at<float>(0, 1) = -sin(Rz);
163  R_z.at<float>(1, 0) = sin(Rz);
164  R_z.at<float>(1, 1) = cos(Rz);
165  // Ry
166  cv::Mat R_y = cv::Mat::eye(3, 3, CV_32F);
167  R_y.at<float>(0, 0) = cos(Ry);
168  R_y.at<float>(0, 2) = sin(Ry);
169  R_y.at<float>(2, 0) = -sin(Ry);
170  R_y.at<float>(2, 2) = cos(Ry);
171  // Rx
172  cv::Mat R_x = cv::Mat::eye(3, 3, CV_32F);
173  R_y.at<float>(1, 1) = cos(Rx);
174  R_y.at<float>(1, 2) = -sin(Rx);
175  R_y.at<float>(2, 1) = sin(Rx);
176  R_y.at<float>(2, 2) = cos(Rx);
177 
178 
179  // Camera Rotation Correction Matrix
180  R = R_z * R_y * R_x;
181 
182  // return rotation matrix
183  return R;
184  }
