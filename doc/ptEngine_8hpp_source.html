149  cv::Mat world2CamRotation() {
150  // camera coordinate to world coordinate rotation matrix
151  cv::Mat R = cv::Mat::zeros(3, 3, CV_32F);
152  // Camera rotation
153  float Rx = degreesToRadians(-90);
154  float Ry = degreesToRadians(0);
155  float Rz = degreesToRadians(-90);
156 
157  // Rz
158  cv::Mat R_z = cv::Mat::eye(3, 3, CV_32F);
159  R_z.at<float>(0, 0) = cos(Rz);
160  R_z.at<float>(0, 1) = -sin(Rz);
161  R_z.at<float>(1, 0) = sin(Rz);
162  R_z.at<float>(1, 1) = cos(Rz);
163  // Ry
164  cv::Mat R_y = cv::Mat::eye(3, 3, CV_32F);
165  R_y.at<float>(0, 0) = cos(Ry);
166  R_y.at<float>(0, 2) = sin(Ry);
167  R_y.at<float>(2, 0) = -sin(Ry);
168  R_y.at<float>(2, 2) = cos(Ry);
169  // Rx
170  cv::Mat R_x = cv::Mat::eye(3, 3, CV_32F);
171  R_y.at<float>(1, 1) = cos(Rx);
172  R_y.at<float>(1, 2) = -sin(Rx);
173  R_y.at<float>(2, 1) = sin(Rx);
174  R_y.at<float>(2, 2) = cos(Rx);
175 
176 
177  // Camera Rotation Correction Matrix
178  R = R_z * R_y * R_x;
179 
180  return R;
181  }
