411 void rodrigues2euler(cv::Mat &R, float &roll, float &pitch, float &yaw) {
412  float cosine_for_pitch = sqrt(pow(R.at<float>(0,0), 2) + pow(R.at<float>(1,0), 2));
413  bool is_singular = false;
414  if(cosine_for_pitch < 10e-6) is_singular = true;
415  if(is_singular == false) {
416  roll = atan2(R.at<float>(2,1), R.at<float>(2,2));
417  pitch = atan2(-R.at<float>(2,0), cosine_for_pitch);
418  yaw = atan2(R.at<float>(1,0), R.at<float>(0,0));
419  }
420  else {
421  roll = 0;
422  pitch = atan2(-R.at<float>(2,0), cosine_for_pitch);
423  yaw = atan2(-R.at<float>(1,2), R.at<float>(1,1));
424  }
