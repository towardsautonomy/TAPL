\subsection*{Installation Instructions}


\begin{DoxyItemize}
\item Download the library.
\end{DoxyItemize}


\begin{DoxyCode}
1 git clone https://github.com/towardsautonomy/TAPL.git
\end{DoxyCode}



\begin{DoxyItemize}
\item Build and install the library as follows.
\end{DoxyItemize}


\begin{DoxyCode}
1 mkdir build  
2 cd build
3 cmake ..
4 make
5 sudo make install
\end{DoxyCode}



\begin{DoxyItemize}
\item Build the examples as follows.
\end{DoxyItemize}


\begin{DoxyCode}
1 cd examples
2 mkdir build
3 cd build
4 cmake ..
5 make
\end{DoxyCode}
